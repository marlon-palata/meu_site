/* 1. Define a animação chamada 'pulsar' */
@keyframes pulsar {
  0% { transform: scale(1); opacity: 1; } /* Estado inicial */
  50% { transform: scale(1.1); opacity: 0.7; } /* Meio da animação */
  100% { transform: scale(1); opacity: 1; } /* Volta ao estado inicial */
}
